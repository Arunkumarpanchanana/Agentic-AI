Why Agents Need Observability

When a production agent fails, you need to answer: which step failed? What was the agent thinking? What tool was called with what arguments? What did the tool return? Without structured observability, debugging agent failures is like investigating a car crash with no dashcam footage — you only see the outcome, not what happened.

Agent observability is harder than traditional software observability because: the "code" is natural language, decisions are probabilistic, and the same input can produce different outcomes on different runs.

The three pillars of agent observability: Traces (step-by-step execution history), Metrics (quantitative performance measures), and Logs (structured event records). LangSmith provides all three out of the box for LangChain/LangGraph agents.

Structured Logging

Plain-text logs are unsearchable at scale. Structured (JSON) logs can be indexed, filtered, and aggregated:

import logging, json, time, uuid
from datetime import datetime

class AgentLogger:
    def __init__(self, agent_id: str, task_id: str):
        self.agent_id = agent_id
        self.task_id = task_id
        self.session_id = str(uuid.uuid4())[:8]
        self._logger = logging.getLogger("agent")

    def _log(self, level: str, event: str, **kwargs):
        record = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": level,
            "agent_id": self.agent_id,
            "task_id": self.task_id,
            "session_id": self.session_id,
            "event": event,
            **kwargs
        }
        print(json.dumps(record))  # In production: send to log aggregator

    def step_start(self, step: int, action: str):
        self._log("INFO", "step_start", step=step, action=action)

    def step_complete(self, step: int, action: str, duration_ms: float, tokens: int):
        self._log("INFO", "step_complete", step=step, action=action,
                  duration_ms=duration_ms, tokens=tokens)

    def tool_call(self, tool: str, args: dict, result_preview: str, success: bool):
        self._log("INFO", "tool_call", tool=tool, args=args,
                  result_preview=result_preview[:100], success=success)

    def error(self, error_type: str, message: str, step: int):
        self._log("ERROR", "agent_error", error_type=error_type,
                  message=message[:500], step=step)

    def task_complete(self, total_steps: int, total_tokens: int, total_cost_usd: float, success: bool):
        self._log("INFO", "task_complete", total_steps=total_steps,
                  total_tokens=total_tokens, total_cost_usd=total_cost_usd, success=success)

logger = AgentLogger("research-agent", "task-001")
logger.step_start(1, "web_search")
logger.tool_call("web_search", {"query": "AI trends 2025"}, "Found 10 results...", True)
logger.step_complete(1, "web_search", duration_ms=342, tokens=450)
logger.task_complete(total_steps=5, total_tokens=2100, total_cost_usd=0.0063, success=True)

Cost Dashboard

from collections import defaultdict
import statistics

class CostTracker:
    # GPT-4o-mini pricing (May 2025)
    PRICING = {"gpt-4o-mini": {"input": 0.15/1e6, "output": 0.60/1e6},
               "gpt-4o":      {"input": 2.50/1e6, "output": 10.00/1e6}}

    def __init__(self):
        self.records = []

    def record(self, model: str, input_tokens: int, output_tokens: int, task_id: str, user_id: str):
        prices = self.PRICING.get(model, self.PRICING["gpt-4o-mini"])
        cost = input_tokens * prices["input"] + output_tokens * prices["output"]
        self.records.append({"model": model, "input": input_tokens, "output": output_tokens,
                              "cost": cost, "task_id": task_id, "user_id": user_id})

    def dashboard(self):
        if not self.records: return
        total_cost = sum(r["cost"] for r in self.records)
        by_user = defaultdict(float)
        for r in self.records: by_user[r["user_id"]] += r["cost"]
        costs = [r["cost"] for r in self.records]
        print(f"=== COST DASHBOARD ===")
        print(f"  Total tasks: {len(self.records)}")
        print(f"  Total cost: ${total_cost:.4f}")
        print(f"  Avg cost/task: ${statistics.mean(costs):.4f}")
        print(f"  Top users by cost:")
        for uid, cost in sorted(by_user.items(), key=lambda x: -x[1])[:5]:
            print(f"    {uid}: ${cost:.4f}")

tracker = CostTracker()
for i in range(10):
    tracker.record("gpt-4o-mini", 1200+i*100, 300+i*50, f"t{i}", f"user-{i%3}")
tracker.dashboard()
Debug workflow: When an agent fails in production: (1) Find the trace in LangSmith. (2) Identify the failing step. (3) Read the exact Thought, Action, and Observation at that step. (4) Reproduce locally with the same inputs. (5) Fix the prompt, tool, or logic. This is why every step must be traced — you cannot debug what you cannot see.

Build a Traced Agent

import time, json, uuid
from dataclasses import dataclass, field, asdict
from typing import Any, Optional

@dataclass
class TraceStep:
    step_id: str = field(default_factory=lambda: str(uuid.uuid4())[:6])
    step_num: int = 0
    action: str = ""
    input: Any = None
    output: Any = None
    duration_ms: float = 0
    tokens: int = 0
    error: Optional[str] = None

@dataclass
class Trace:
    trace_id: str = field(default_factory=lambda: str(uuid.uuid4())[:8])
    goal: str = ""
    steps: list = field(default_factory=list)
    success: bool = False
    total_tokens: int = 0
    total_cost_usd: float = 0
    started_at: float = field(default_factory=time.time)

class TracedAgent:
    TOKEN_COST = 0.15 / 1e6

    def __init__(self, name: str):
        self.name = name
        self.traces = []

    def run(self, goal: str, inject_failure_at: int = None) -> Trace:
        trace = Trace(goal=goal)
        steps_plan = [
            ("search", f"search({goal[:30]})"),
            ("analyse", "analyse(results)"),
            ("synthesise", "synthesise(analysis)"),
            ("format", "format_output()"),
        ]
        for i, (action, description) in enumerate(steps_plan):
            step = TraceStep(step_num=i+1, action=action, input=description)
            t0 = time.time()
            if inject_failure_at == i+1:
                step.error = f"Simulated failure at step {i+1}: API timeout"
                step.duration_ms = round((time.time()-t0)*1000, 1)
                trace.steps.append(step)
                self._print_step(step)
                break
            time.sleep(0.05)
            tokens = 200 + i * 50
            step.output = f"Result of {action}"
            step.duration_ms = round((time.time()-t0)*1000+50, 1)
            step.tokens = tokens
            trace.total_tokens += tokens
            trace.steps.append(step)
            self._print_step(step)
        trace.success = not any(s.error for s in trace.steps) and len(trace.steps) == len(steps_plan)
        trace.total_cost_usd = round(trace.total_tokens * self.TOKEN_COST, 6)
        self.traces.append(trace)
        return trace

    def _print_step(self, s: TraceStep):
        status = "❌ ERROR" if s.error else "✅"
        print(f"  {status} Step {s.step_num} [{s.action}] {s.duration_ms:.0f}ms {s.tokens} tok")
        if s.error: print(f"     Error: {s.error}")

agent = TracedAgent("research-agent")
print("=== Successful run ===")
t1 = agent.run("Research quantum computing advances")

print("\n=== Failed run (injected failure at step 3) ===")

def inspect_trace(trace: Trace):
    print(f"\n=== TRACE INSPECTION: {trace.trace_id} ===")
    print(f"  Goal: {trace.goal}")
    print(f"  Success: {trace.success}")
    print(f"  Steps completed: {len(trace.steps)}")
    print(f"  Total tokens: {trace.total_tokens}")
    print(f"  Total cost: ${trace.total_cost_usd:.6f}")
    failing = [s for s in trace.steps if s.error]
    if failing:
        print(f"\n  FAILURE ANALYSIS:")
        for s in failing:
            print(f"    Step {s.step_num} [{s.action}] failed: {s.error}")
            print(f"    Input was: {s.input}")
            completed = [st for st in trace.steps if not st.error]
            print(f"    {len(completed)} steps completed before failure")

inspect_trace(t2)

Aggregate Metrics Across Runs

import statistics

# Run 5 more tasks with mixed results
for i in range(5):
    fail_at = 2 if i == 1 else (4 if i == 3 else None)
    agent.run(f"Task {i+3}", inject_failure_at=fail_at)

def metrics_report(agent):
    traces = agent.traces
    successful = [t for t in traces if t.success]
    failed = [t for t in traces if not t.success]
    all_steps = [len(t.steps) for t in traces]
    all_costs = [t.total_cost_usd for t in traces]
    print(f"\n=== METRICS REPORT ({len(traces)} tasks) ===")
    print(f"  Success rate: {len(successful)/len(traces):.0%}")
    print(f"  Failure rate: {len(failed)/len(traces):.0%}")
    print(f"  Avg steps/task: {statistics.mean(all_steps):.1f}")
    print(f"  Total cost: ${sum(all_costs):.4f}")
    print(f"  Avg cost/task: ${statistics.mean(all_costs):.5f}")
    if successful:
        print(f"  Avg cost (success only): ${statistics.mean(t.total_cost_usd for t in successful):.5f}")

metrics_report(agent)
