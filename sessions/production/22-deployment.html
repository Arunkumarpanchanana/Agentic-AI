Production Challenges for Agents

Running an agent in a notebook is easy. Running it reliably in production for thousands of users is hard. Production agents must handle: concurrent requests, long-running tasks (minutes to hours), failures and retries, persistent state across restarts, cost control, and observability. Each of these requires deliberate engineering decisions.

Production stack: FastAPI (REST API) + Celery (async task queue) + Redis (message broker + result cache) + PostgreSQL (persistent state) + Docker (containerization) + Prometheus/Grafana (monitoring).

FastAPI Agent Wrapper

Wrap your agent in a FastAPI application that accepts task requests, queues them for async execution, and returns a task ID for polling:

from fastapi import FastAPI, BackgroundTasks, HTTPException
from pydantic import BaseModel
from typing import Optional
import uuid, asyncio, time
from datetime import datetime

app = FastAPI(title="Agent API", version="1.0.0")

# In-memory task store (replace with Redis/DB in production)
tasks: dict = {}

class TaskRequest(BaseModel):
    goal: str
    user_id: str
    priority: int = 1  # 1 (normal) to 5 (urgent)
    timeout_seconds: int = 300

class TaskResponse(BaseModel):
    task_id: str
    status: str
    created_at: str
    result: Optional[str] = None
    error: Optional[str] = None

async def run_agent_task(task_id: str, goal: str, timeout: int):
    """Background coroutine that runs the agent."""
    tasks[task_id]["status"] = "running"
    try:
        # Simulate agent execution
        await asyncio.sleep(2)  # Replace with actual agent.run(goal)
        tasks[task_id].update({
            "status": "completed",
            "result": f"Completed: {goal[:50]}",
            "completed_at": datetime.now().isoformat()
        })
    except asyncio.TimeoutError:
        tasks[task_id].update({"status": "timeout", "error": f"Exceeded {timeout}s"})
    except Exception as e:
        tasks[task_id].update({"status": "failed", "error": str(e)[:500]})

@app.post("/tasks", response_model=TaskResponse)
async def create_task(req: TaskRequest, background_tasks: BackgroundTasks):
    task_id = str(uuid.uuid4())
    tasks[task_id] = {
        "task_id": task_id, "status": "queued",
        "goal": req.goal, "user_id": req.user_id,
        "created_at": datetime.now().isoformat()
    }
    background_tasks.add_task(run_agent_task, task_id, req.goal, req.timeout_seconds)
    return TaskResponse(**tasks[task_id])

@app.get("/tasks/{task_id}", response_model=TaskResponse)
async def get_task(task_id: str):
    if task_id not in tasks:
        raise HTTPException(status_code=404, detail="Task not found")
    return TaskResponse(**tasks[task_id])

@app.get("/health")
async def health_check():
    return {"status": "healthy", "timestamp": datetime.now().isoformat(),
            "active_tasks": sum(1 for t in tasks.values() if t["status"] == "running"),
            "queued_tasks": sum(1 for t in tasks.values() if t["status"] == "queued")}

Celery for Async Task Queues

# celery_app.py
from celery import Celery
from kombu import Queue

celery_app = Celery(
    "agent_tasks",
    broker="redis://localhost:6379/0",
    backend="redis://localhost:6379/1",
)

celery_app.conf.update(
    task_queues=[
        Queue("high_priority", routing_key="high"),
        Queue("normal",        routing_key="normal"),
    ],
    task_default_queue="normal",
    task_time_limit=600,         # Hard limit: kill after 10 min
    task_soft_time_limit=540,    # Soft limit: raise exception at 9 min
    task_acks_late=True,         # Only ack after successful completion
    task_reject_on_worker_lost=True,  # Requeue if worker crashes
    result_expires=3600,         # Keep results for 1 hour
)

@celery_app.task(bind=True, max_retries=3, default_retry_delay=30)
def run_agent(self, goal: str, user_id: str) -> dict:
    try:
        # Import agent here to avoid circular imports
        from my_agent import agent
        result = agent.run(goal)
        return {"status": "completed", "result": result, "user_id": user_id}
    except Exception as e:
        # Retry with exponential backoff
        raise self.retry(exc=e, countdown=2 ** self.request.retries * 30)

Production checklist: (1) All secrets in environment variables, never in code. (2) Task timeouts set on both API and worker sides. (3) Retry logic with exponential backoff. (4) Health check endpoint for load balancer. (5) Structured logging with correlation IDs. (6) Cost caps per user/task.

%%writefile main.py
from fastapi import FastAPI, BackgroundTasks
from pydantic import BaseModel
from typing import Optional
import uuid, asyncio
from datetime import datetime

app = FastAPI()
tasks = {}

class TaskReq(BaseModel):
    goal: str

@app.post("/run")
async def run_task(req: TaskReq, bg: BackgroundTasks):
    tid = str(uuid.uuid4())[:8]
    tasks[tid] = {"status": "queued", "created": datetime.now().isoformat()}
    async def execute():
        tasks[tid]["status"] = "running"
        await asyncio.sleep(1)  # Simulate agent
        tasks[tid].update({"status": "done", "result": f"Completed: {req.goal[:40]}"})
    bg.add_task(execute)
    return {"task_id": tid, "status": "queued"}

@app.get("/tasks/{tid}")
async def get_task(tid: str):
    return tasks.get(tid, {"error": "not found"})

@app.get("/health")
def health(): return {"status": "ok", "tasks": len(tasks)}

import requests, time

BASE = "http://localhost:8000"

# Submit a task
r = requests.post(f"{BASE}/run", json={"goal": "Research the latest AI news"})
tid = r.json()["task_id"]
print(f"Task submitted: {tid}")

# Poll for completion
for _ in range(10):
    time.sleep(0.5)
    status = requests.get(f"{BASE}/tasks/{tid}").json()
    print(f"  Status: {status.get('status')}")
    if status.get("status") == "done":
        print(f"  Result: {status.get('result')}")
        break

# Health check
health = requests.get(f"{BASE}/health").json()
print(f"\nHealth: {health}")

# Cleanup
import subprocess
subprocess.run(["docker", "stop", "agent-api"], capture_output=True)
subprocess.run(["docker", "rm", "agent-api"], capture_output=True)
print("Container cleaned up")
