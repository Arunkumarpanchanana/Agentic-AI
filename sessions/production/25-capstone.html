Project Track A: Autonomous Research Agent

Autonomous Research Agent

Given a research question, the agent autonomously searches the web, retrieves and reads papers, extracts key findings, cross-checks facts across sources, and generates a structured research report with citations.

WebSearch RAG Reflection Multi-source synthesis

Core components: ReAct planner → search tool → web scraper → ChromaDB memory → critic agent → report generator → hallucination checker → human review gate.

from langchain_openai import ChatOpenAI
from langchain.agents import AgentExecutor, create_react_agent
from langchain_community.tools import DuckDuckGoSearchRun, WikipediaQueryRun
from langchain_community.utilities import WikipediaAPIWrapper
from langchain.memory import ConversationSummaryBufferMemory
from langchain.prompts import PromptTemplate

RESEARCH_PROMPT = PromptTemplate.from_template("""
You are an autonomous research agent. Your goal is to answer the research question
thoroughly, with evidence and citations.

Research Question: {input}

You have access to the following tools:
{tools}

Use this exact format:
Thought: what I need to find next
Action: tool_name
Action Input: query
Observation: tool result
... (repeat as needed, at least 3 searches)
Thought: I have enough information
Final Answer: [structured report with: Summary, Key Findings (bullet list), Citations]

{agent_scratchpad}
""")

def build_research_agent():
    llm = ChatOpenAI(model="gpt-4o", temperature=0.1)
    tools = [
        DuckDuckGoSearchRun(name="web_search"),
        WikipediaQueryRun(api_wrapper=WikipediaAPIWrapper(top_k_results=2), name="wikipedia"),
    ]
    memory = ConversationSummaryBufferMemory(
        llm=llm, max_token_limit=2000, return_messages=True
    )
    agent = create_react_agent(llm, tools, RESEARCH_PROMPT)
    return AgentExecutor(agent=agent, tools=tools, memory=memory, verbose=True, max_iterations=12)

Project Track B: Coding Assistant Agent

Coding Assistant Agent

Given a natural-language feature request or bug description, the agent reads existing code, writes or patches functions, runs tests in a sandbox, inspects failures, and iterates until tests pass — with a human approval gate before any file is written.

Code execution Test-driven Human-in-the-loop Reflexion

Core components: Plan step → code generator → sandbox executor → test runner → failure analyser (Reflexion) → approval gate → file writer.

import subprocess, tempfile, os

def execute_python_safely(code: str, timeout: int = 10) -> dict:
    """Run Python code in a subprocess sandbox."""
    with tempfile.NamedTemporaryFile(mode="w", suffix=".py", delete=False) as f:
        f.write(code)
        path = f.name
    try:
        result = subprocess.run(
            ["python3", path],
            capture_output=True, text=True, timeout=timeout
        )
        return {
            "stdout": result.stdout,
            "stderr": result.stderr,
            "returncode": result.returncode,
            "success": result.returncode == 0
        }
    except subprocess.TimeoutExpired:
        return {"stdout": "", "stderr": "Timeout", "returncode": -1, "success": False}
    finally:
        os.unlink(path)

def coding_agent_loop(llm, task: str, tests: str, max_iterations: int = 5) -> str:
    history = []
    for iteration in range(max_iterations):
        prompt = f"""Task: {task}
Tests that must pass:
{tests}
Previous attempts: {history[-2:] if history else 'None'}
Write only the Python implementation (no markdown fences)."""
        code = llm.invoke(prompt).content
        result = execute_python_safely(tests.replace("# IMPL\n", code + "\n"))
        if result["success"]:
            print(f"Tests passed on iteration {iteration + 1}")
            return code
        # Reflexion: critique the failure
        critique = llm.invoke(
            f"The code failed with:\n{result['stderr']}\nCode was:\n{code}\n"
            f"Identify the bug in one sentence."
        ).content
        history.append({"code": code[:200], "error": result["stderr"][:200], "critique": critique})
        print(f"Iteration {iteration+1} failed: {critique[:80]}")
    return "Max iterations reached — manual review needed"

Project Track C: Data Analyst Agent

Data Analyst Agent

Given a CSV or database connection and a business question, the agent explores the schema, generates and executes SQL/pandas queries, interprets results, produces visualisations, and writes an executive summary — with self-consistency checks on key statistics.

Pandas/SQL Code execution Self-consistency Visualisation

Core components: Schema explorer → query planner → code executor → chart generator → self-consistency verifier (run key queries 3×) → insight extractor → report writer.

import pandas as pd
import io

def data_analyst_agent(llm, csv_path: str, question: str) -> dict:
    df = pd.read_csv(csv_path)
    schema = f"Columns: {list(df.columns)}\nDtypes:\n{df.dtypes.to_string()}\nShape: {df.shape}\nSample:\n{df.head(3).to_string()}"

    # Step 1: plan the analysis
    plan = llm.invoke(
        f"Dataset schema:\n{schema}\n\nQuestion: {question}\n\n"
        "Write a numbered analysis plan (3–5 steps). Be concise."
    ).content
    print("Plan:", plan)

    # Step 2: generate pandas code
    code = llm.invoke(
        f"Dataset schema:\n{schema}\n\nQuestion: {question}\n\n"
        "Write Python pandas code that answers the question. "
        "The dataframe is already loaded as 'df'. "
        "Print the key result as the last line. No imports needed."
    ).content.strip().strip("```python").strip("```")

    # Step 3: execute
    exec_globals = {"df": df, "pd": pd}
    try:
        exec(code, exec_globals)
        # Step 4: interpret
        insight = llm.invoke(
            f"Question: {question}\nAnalysis code executed successfully.\n"
            f"Write a 2-sentence executive summary of the findings."
        ).content
        return {"status": "success", "plan": plan, "code": code, "insight": insight}
    except Exception as e:
        return {"status": "error", "error": str(e), "code": code}

Implementation Phase — Core Agent Loop

Implement your agent using the starter scaffold for your chosen track. Add semantic caching, token budgeting, and parallel tool calls where applicable.

import time, random

class CapstoneAgent:
    """
    Generic scaffold — extend for your chosen track.
    Replace the mock LLM and tools with real implementations.
    """
    def __init__(self, design: dict):
        self.design = design
        self.token_budget = 8000
        self.tokens_used = 0
        self.cache = {}   # simple string cache for demo
        self.trace = []

    def _mock_llm(self, prompt: str, model: str = "gpt-4o") -> str:
        cost_map = {"gpt-4o": 20, "gpt-4o-mini": 1}
        self.tokens_used += cost_map.get(model, 10)
        if self.tokens_used > self.token_budget:
            raise RuntimeError(f"Token budget exceeded: {self.tokens_used}")
        return f"[{model}] Response to: {prompt[:60]}..."

    def _mock_tool(self, name: str, query: str) -> str:
        time.sleep(0.1)  # simulate API latency
        return f"Tool '{name}' result for '{query}'"

    def run(self, task: str) -> dict:
        start = time.time()
        self.trace = []

        # Step 1: plan
        plan = self._mock_llm(f"Plan how to: {task}", model="gpt-4o")
        self.trace.append({"step": "plan", "output": plan})

        # Step 2: check cache
        if task in self.cache:
            self.trace.append({"step": "cache_hit"})
            return {"result": self.cache[task], "from_cache": True, "tokens": 0}

        # Step 3: execute tools (parallel in real impl)
        tool_results = {}
        for tool in self.design["tools"]:
            tool_results[tool] = self._mock_tool(tool, task)
        self.trace.append({"step": "tools", "results": tool_results})

        # Step 4: synthesise
        answer = self._mock_llm(
            f"Synthesise: {task}\nEvidence: {tool_results}",
            model="gpt-4o"
        )
        self.trace.append({"step": "synthesise", "output": answer})

        # Step 5: self-critique
        critique = self._mock_llm(
            f"Critique this answer: {answer}",
            model="gpt-4o-mini"
        )
        self.trace.append({"step": "critique", "output": critique})

        self.cache[task] = answer
        latency = time.time() - start

        return {
            "result": answer,
            "from_cache": False,
            "tokens_used": self.tokens_used,
            "latency_s": round(latency, 2),
            "steps": len(self.trace),
        }

agent = CapstoneAgent(AGENT_DESIGN)
result = agent.run("What are the main approaches to AI alignment?")

def evaluate_agent(agent: CapstoneAgent, tasks: list) -> dict:
    results = []
    for task in tasks:
        try:
            r = agent.run(task)
            results.append({
                "task": task[:50],
                "success": True,
                "tokens": r["tokens_used"],
                "latency": r["latency_s"],
            })
        except Exception as e:
            results.append({
                "task": task[:50],
                "success": False,
                "error": str(e),
                "tokens": 0,
                "latency": 0,
            })

    successful = [r for r in results if r["success"]]
    success_rate = len(successful) / len(results)
    avg_latency = sum(r["latency"] for r in successful) / max(len(successful), 1)
    total_tokens = sum(r["tokens"] for r in results)
    est_cost = total_tokens * 5e-6  # rough GPT-4o estimate

    report = {
        "total_tasks": len(results),
        "success_rate": f"{success_rate:.0%}",
        "avg_latency_s": round(avg_latency, 2),
        "total_tokens": total_tokens,
        "estimated_cost_usd": round(est_cost, 4),
        "passed": success_rate >= 0.80,
    }
    print("\n=== EVALUATION REPORT ===")
    for k, v in report.items():
        print(f"  {k}: {v}")
    return report

Ship It — FastAPI Endpoint

Wrap your agent in a FastAPI endpoint and verify it responds correctly. This is the last step before production deployment.

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import uvicorn

app = FastAPI(title="Agentic AI Capstone", version="1.0")
_agent = CapstoneAgent(AGENT_DESIGN)

class TaskRequest(BaseModel):
    task: str
    track: str = "research"

class TaskResponse(BaseModel):
    result: str
    tokens_used: int
    latency_s: float
    from_cache: bool
    steps: int

@app.post("/run", response_model=TaskResponse)
async def run_task(req: TaskRequest):
    try:
        result = _agent.run(req.task)
        return TaskResponse(**result)
    except RuntimeError as e:
        raise HTTPException(status_code=429, detail=str(e))

@app.get("/health")
def health():
    return {"status": "ok", "tokens_used": _agent.tokens_used}

@app.get("/metrics")
def metrics():
    return {
        "cache_size": len(_agent.cache),
        "tokens_used": _agent.tokens_used,
        "token_budget": _agent.token_budget,
        "budget_remaining_pct": round(
            (1 - _agent.tokens_used / _agent.token_budget) * 100, 1
        ),
    }
