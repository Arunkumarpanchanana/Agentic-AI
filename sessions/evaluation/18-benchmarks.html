Agent Benchmarks & Metrics

WebArena, AgentBench, GAIA, SWE-Bench — and how to design your own evaluation harness for production agents.

Why Benchmarks Matter

Without rigorous evaluation, agent development is guesswork. A benchmark provides a standardized, reproducible measure of agent capability. It tells you: is version 2 actually better than version 1? Does the agent generalize beyond the tasks it was prompted with? Where exactly does it fail?

Agent benchmarks are harder to design than model benchmarks because agents take sequences of actions — partial credit, efficiency, and reliability across runs all matter, not just the final answer.

The four key metrics: Task success rate (did it complete?), efficiency (steps and tokens used), reliability (variance across runs), and cost (total spend). Optimizing one often trades off against others.

The Four Core Metrics in Detail

1. Task Success Rate (TSR)

The fraction of tasks completed successfully. The most important metric, but requires careful definition of "success." Partial credit rubrics handle tasks where the agent gets partway there.

def task_success_rate(results: list) -> dict:
    total = len(results)
    full_success = sum(1 for r in results if r["score"] == 1.0)
    partial = sum(1 for r in results if 0 < r["score"] < 1.0)
    failed = sum(1 for r in results if r["score"] == 0)
    return {
        "tsr": full_success / total,
        "partial_rate": partial / total,
        "failure_rate": failed / total,
        "mean_score": sum(r["score"] for r in results) / total
    }

2. Efficiency (Steps & Tokens)

How many steps and tokens did the agent use to complete a task? An agent that completes a task in 3 steps is better than one that takes 15 steps — it costs less and is less likely to drift.

def efficiency_metrics(results: list) -> dict:
    successful = [r for r in results if r["score"] == 1.0]
    if not successful: return {"avg_steps": None, "avg_tokens": None}
    return {
        "avg_steps": sum(r["steps"] for r in successful) / len(successful),
        "avg_tokens": sum(r["tokens"] for r in successful) / len(successful),
        "steps_per_success": sum(r["steps"] for r in results) / max(1, sum(1 for r in results if r["score"]==1.0)),
    }

3. Reliability (Variance Across Runs)

Run each task 3-5 times. A reliable agent gets the same result consistently. High variance is a red flag — the agent is essentially guessing.

import statistics

def reliability_metrics(repeated_results: dict) -> dict:
    """repeated_results: {task_id: [score_run1, score_run2, score_run3]}"""
    variances = {tid: statistics.variance(scores) for tid, scores in repeated_results.items() if len(scores) > 1}
    return {
        "mean_variance": statistics.mean(variances.values()) if variances else 0,
        "high_variance_tasks": [tid for tid, v in variances.items() if v > 0.25],
        "fully_reliable_rate": sum(1 for scores in repeated_results.values() if len(set(scores))==1) / len(repeated_results)
    }

Building a Test Harness

import time, json
from dataclasses import dataclass, field, asdict

@dataclass
class EvalResult:
    task_id: str
    task_description: str
    expected_output: str
    actual_output: str
    score: float       # 0.0 - 1.0
    steps: int
    tokens_used: int
    duration_seconds: float
    error: str = ""

class AgentEvaluator:
    def __init__(self, agent, scorer_fn, test_cases: list):
        self.agent = agent
        self.scorer = scorer_fn
        self.test_cases = test_cases
        self.results: list[EvalResult] = []

    def run(self) -> list[EvalResult]:
        for tc in self.test_cases:
            start = time.time()
            try:
                output = self.agent.run(tc["input"])
                score = self.scorer(output, tc["expected"])
                error = ""
            except Exception as e:
                output, score, error = "", 0.0, str(e)[:200]
            self.results.append(EvalResult(
                task_id=tc["id"], task_description=tc["input"],
                expected_output=tc["expected"], actual_output=str(output),
                score=score, steps=getattr(self.agent, "last_steps", 0),
                tokens_used=getattr(self.agent, "last_tokens", 0),
                duration_seconds=round(time.time()-start, 2), error=error
            ))
        return self.results

    def report(self):
        tsr = sum(r.score for r in self.results) / len(self.results)
        avg_steps = sum(r.steps for r in self.results) / len(self.results)
        print(f"=== EVAL REPORT ({len(self.results)} tasks) ===")
        print(f"  Task Success Rate: {tsr:.1%}")
        print(f"  Avg Steps: {avg_steps:.1f}")
        print(f"  Failures: {sum(1 for r in self.results if r.score==0)}")
        for r in self.results:
            status = "✅" if r.score == 1 else "⚠️" if r.score > 0 else "❌"
            print(f"  {status} {r.task_id}: score={r.score:.1f}, steps={r.steps}")

2

Build the Scorer

def score_code_output(output: str, test_case: dict) -> float:
    """Multi-dimensional scorer for code generation tasks."""
    score = 0.0
    keywords = test_case.get("expected_contains", [])
    if not keywords: return 0.5
    # Keyword presence (50% of score)
    found = sum(1 for kw in keywords if kw in output)
    score += 0.5 * (found / len(keywords))
    # Syntactic validity (30% of score)
    try:
        compile(output, "", "exec")
        score += 0.3
    except SyntaxError:
        pass
    # Has docstring or comments (20%)
    if '"""' in output or "'''" in output or "#" in output:
        score += 0.2
    return round(min(score, 1.0), 2)

# Test scorer
sample_output = '''
def reverse_string(s):
    """Reverse a string."""
    return s[::-1]
'''
tc = TEST_CASES[0]
print(f"Score for tc-01: {score_code_output(sample_output, tc)}")
3

Run the Evaluation Harness

from langchain_openai import ChatOpenAI
from langchain.schema import HumanMessage
import time, statistics

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

results = []
for tc in TEST_CASES:
    start = time.time()
    response = llm([HumanMessage(content=f"Write only Python code, no explanation:\n{tc['input']}")]).content
    duration = time.time() - start
    score = score_code_output(response, tc)
    results.append({"id": tc["id"], "score": score, "duration": round(duration,2), "difficulty": tc["difficulty"]})
    print(f"  [{tc['id']}] score={score:.2f}, time={duration:.1f}s")

# Aggregate metrics
print(f"\n=== RESULTS ===")
print(f"  Mean score: {statistics.mean(r['score'] for r in results):.2f}")
print(f"  Pass rate (>=0.7): {sum(1 for r in results if r['score']>=0.7)/len(results):.0%}")
print(f"  Avg duration: {statistics.mean(r['duration'] for r in results):.1f}s")

Analyze by Difficulty

from collections import defaultdict

by_difficulty = defaultdict(list)
for r in results:
    by_difficulty[r["difficulty"]].append(r["score"])

print("Performance by difficulty:")
for diff in sorted(by_difficulty.keys()):
    scores = by_difficulty[diff]
    avg = statistics.mean(scores)
    print(f"  Difficulty {diff}: avg={avg:.2f}, tasks={len(scores)}")
