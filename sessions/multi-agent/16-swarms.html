Swarm Intelligence

Swarm intelligence is the collective behaviour that emerges when many simple agents interact locally, producing complex, adaptive, globally intelligent behaviour without central control. Nature's examples: ant colonies finding shortest paths, bird flocking, bee colony decision-making. These systems are robust, scalable, and self-organizing — all properties valuable in AI agent systems.

Key insight: The intelligence is in the system, not the individual. Each agent follows simple local rules. Complex global behaviour emerges from their interactions. This means you can build sophisticated document processing pipelines from five simple, specialized micro-agents.

Stigmergy: Coordination Through the Environment

Stigmergy is indirect coordination through environmental modification. Ants don't communicate directly — they deposit pheromones that other ants follow, creating self-reinforcing trails to food sources. In agent systems, stigmergy is implemented through a shared state that agents modify and read.

import time, random
from collections import defaultdict

class StigmergyBoard:
    """Shared environment that agents modify to coordinate."""
    def __init__(self, evaporation_rate=0.05):
        self.pheromones = defaultdict(float)  # path -> strength
        self.evaporation_rate = evaporation_rate

    def deposit(self, path: str, amount: float = 1.0):
        self.pheromones[path] += amount
        # Reinforce paths proportional to quality

    def evaporate(self):
        for path in list(self.pheromones.keys()):
            self.pheromones[path] *= (1 - self.evaporation_rate)
            if self.pheromones[path] < 0.01:
                del self.pheromones[path]

    def best_path(self) -> str | None:
        if not self.pheromones: return None
        return max(self.pheromones, key=self.pheromones.get)

class AntAgent:
    def __init__(self, agent_id: str, board: StigmergyBoard):
        self.id = agent_id
        self.board = board
        self.paths = ["path_A", "path_B", "path_C"]

    def explore(self) -> str:
        """Choose path probabilistically based on pheromone strength."""
        weights = [self.board.pheromones.get(p, 0.1) for p in self.paths]
        total = sum(weights)
        probs = [w/total for w in weights]
        chosen = random.choices(self.paths, weights=probs)[0]
        # Simulate path quality (path_A is best)
        quality = {"path_A": 1.5, "path_B": 0.8, "path_C": 0.4}[chosen]
        self.board.deposit(chosen, quality)
        return chosen

board = StigmergyBoard()
ants = [AntAgent(f"ant-{i}", board) for i in range(10)]
print("Running ant colony simulation (20 rounds):")
for round_num in range(20):
    choices = [ant.explore() for ant in ants]
    board.evaporate()
    if round_num % 5 == 4:
        print(f"  Round {round_num+1}: pheromones = {dict(board.pheromones)}")
print(f"\nEmergent best path: {board.best_path()}")

Boid Rules for Agent Flocking

Craig Reynolds' boid model (1986) generates realistic flocking from three simple local rules:

  • Separation: Avoid crowding neighbors — maintain minimum distance.
  • Alignment: Steer toward the average heading of neighbors.
  • Cohesion: Steer toward the average position of neighbors.

In LLM agent systems, these map to: separation = don't duplicate work other agents are doing; alignment = adopt the consensus approach of the group; cohesion = stay focused on the shared goal.

Practical application: The most immediately useful swarm idea for LLM agents is stigmergy via a shared blackboard. Instead of explicit orchestration, agents mark what they've done and what needs doing. New agents check the board and pick up available work. This creates a self-organizing pipeline with no single orchestrator.

Define the Shared Blackboard

import time, random
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class WorkItem:
    item_id: str
    content: str
    item_type: str       # "raw_chunk", "summary", "fact", "final"
    status: str = "available"  # "available", "claimed", "done"
    result: Optional[str] = None
    claimed_by: Optional[str] = None
    score: float = 0.0   # quality/confidence score (stigmergy)

class Blackboard:
    def __init__(self):
        self._items: dict[str, WorkItem] = {}

    def post(self, item: WorkItem):
        self._items[item.item_id] = item

    def claim(self, item_type: str, agent_id: str) -> Optional[WorkItem]:
        """Atomically claim an available item of given type."""
        available = [i for i in self._items.values()
                     if i.item_type == item_type and i.status == "available"]
        if not available: return None
        item = random.choice(available)
        item.status = "claimed"
        item.claimed_by = agent_id
        return item

    def complete(self, item_id: str, result: str, score: float = 1.0):
        item = self._items[item_id]
        item.status = "done"
        item.result = result
        item.score = score

    def get_done(self, item_type: str) -> list:
        return [i for i in self._items.values() if i.item_type == item_type and i.status == "done"]

    def summary(self):
        from collections import Counter
        counts = Counter(f"{i.item_type}:{i.status}" for i in self._items.values())
        for k, v in sorted(counts.items()): print(f"  {k}: {v}")

Create Specialized Micro-Agents

class SummarizerAgent:
    """Summarizes raw chunks into concise summaries."""
    def __init__(self, agent_id, bb): self.id, self.bb = agent_id, bb
    def run(self):
        item = self.bb.claim("raw_chunk", self.id)
        if not item: return False
        summary = f"[Summary by {self.id}] {item.content[:60]}... → Key point extracted."
        result_item = WorkItem(f"sum-{item.item_id}", summary, "summary")
        self.bb.post(result_item)
        self.bb.complete(item.item_id, summary, score=0.8)
        print(f"  {self.id}: summarized {item.item_id}")
        return True

class FactExtractorAgent:
    """Extracts key facts from summaries."""
    def __init__(self, agent_id, bb): self.id, self.bb = agent_id, bb
    def run(self):
        item = self.bb.claim("summary", self.id)
        if not item: return False
        fact = f"FACT: {item.content.split('→')[-1].strip()}"
        result_item = WorkItem(f"fact-{item.item_id}", fact, "fact")
        self.bb.post(result_item)
        self.bb.complete(item.item_id, fact, score=0.9)
        print(f"  {self.id}: extracted fact from {item.item_id}")
        return True

class QualityAgent:
    """Scores facts for quality (stigmergy: deposits confidence)."""
    def __init__(self, agent_id, bb): self.id, self.bb = agent_id, bb
    def run(self):
        facts = self.bb.get_done("fact")
        unscored = [f for f in facts if f.score < 0.5]
        if not unscored: return False
        fact = random.choice(unscored)
        fact.score = random.uniform(0.6, 1.0)  # score it
        print(f"  {self.id}: scored fact {fact.item_id} → {fact.score:.2f}")
        return True

agents = [SummarizerAgent("sum-1", bb), SummarizerAgent("sum-2", bb),
          FactExtractorAgent("fact-1", bb), QualityAgent("quality-1", bb)]
print("\nAgents created:", [a.id for a in agents])

Synthesizer Agent Collects High-Quality Facts

class SynthesizerAgent:
    """Collects high-confidence facts and produces final summary."""
    def __init__(self, bb): self.bb = bb
    def run(self, min_score=0.7):
        facts = self.bb.get_done("fact")
        high_quality = [f for f in facts if f.score >= min_score]
        high_quality.sort(key=lambda f: -f.score)
        print(f"\n=== FINAL SYNTHESIS ({len(high_quality)}/{len(facts)} facts meet quality threshold) ===")
        for f in high_quality:
            print(f"  [{f.score:.2f}] {f.content[:80]}")
        synthesis = f"Document Summary: {len(high_quality)} key facts extracted about quantum computing including superposition, entanglement, algorithms, and hardware."
        self.bb.post(WorkItem("final", synthesis, "final"))
        self.bb.complete("final", synthesis, score=1.0)
        return synthesis

synth = SynthesizerAgent(bb)
final = synth.run()
print(f"\nFinal output: {final}")
