Design approval gates, escalation policies, and interrupt signals that keep humans appropriately in control.

When Should an Agent Pause and Ask?

Full autonomy is not always desirable — or safe. The art of human-in-the-loop design is knowing exactly when to interrupt the agent and ask for human input, without interrupting so often that the agent becomes annoying and useless.

Three conditions warrant a pause:

  1. Ambiguity: The agent cannot determine the user's intent from available context. Rather than guess wrong, it should ask. Example: "Send the report to the team" — which team? Which report?
  2. Risk: The action has a high potential for harm if wrong. Example: deleting files, sending external emails, making financial transactions.
  3. Irreversibility: The action cannot be undone. Example: publishing a blog post, deploying to production, sending a mass notification.
The minimal footprint principle: Agents should request only necessary permissions, prefer reversible actions over irreversible ones, and err on the side of doing less and confirming with users when uncertain about intended scope.

Escalation Policies

Not every pause needs to go to the same human. A well-designed escalation policy routes based on urgency and domain:

from enum import Enum
from dataclasses import dataclass

class RiskLevel(Enum):
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    CRITICAL = 4

@dataclass
class Action:
    name: str
    description: str
    risk_level: RiskLevel
    reversible: bool

ESCALATION_POLICY = {
    RiskLevel.LOW:      {"approver": "auto",          "timeout_minutes": 0,  "fallback": "skip"},
    RiskLevel.MEDIUM:   {"approver": "user",           "timeout_minutes": 60, "fallback": "skip"},
    RiskLevel.HIGH:     {"approver": "user",           "timeout_minutes": 10, "fallback": "abort"},
    RiskLevel.CRITICAL: {"approver": "security_team",  "timeout_minutes": 5,  "fallback": "abort"},
}

def should_pause(action: Action) -> bool:
    policy = ESCALATION_POLICY[action.risk_level]
    if policy["approver"] == "auto":
        return False
    if not action.reversible and action.risk_level.value >= RiskLevel.MEDIUM.value:
        return True
    return policy["approver"] != "auto"

# Examples
actions = [
    Action("search_web", "Search Google", RiskLevel.LOW, True),
    Action("draft_email", "Draft an email response", RiskLevel.MEDIUM, True),
    Action("send_email", "Send email to 500 customers", RiskLevel.HIGH, False),
    Action("drop_table", "Delete production database table", RiskLevel.CRITICAL, False),
]
for a in actions:
    pause = should_pause(a)
    print(f"{'🛑 PAUSE' if pause else '✅ AUTO'} — {a.name} (risk={a.risk_level.name}, reversible={a.reversible})")

LangGraph Human Approval Node

from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
from typing import TypedDict, Annotated
import operator

class AgentState(TypedDict):
    messages: list
    pending_action: dict
    approved: bool
    final_output: str

def plan_action(state: AgentState) -> AgentState:
    """Agent decides what action to take."""
    state["pending_action"] = {
        "type": "send_email",
        "to": "customer@example.com",
        "subject": "Your order update",
        "body": "Your order #1234 has shipped."
    }
    return state

def human_approval(state: AgentState) -> AgentState:
    """Human-in-the-loop node — in production this would send a notification."""
    action = state["pending_action"]
    print(f"\n[APPROVAL REQUIRED]")
    print(f"  Action: {action['type']}")
    print(f"  To: {action.get('to')}")
    print(f"  Subject: {action.get('subject')}")
    # In a real system: send Slack message, email, or UI notification
    # Here we simulate approval
    state["approved"] = True  # or False to reject
    return state

def execute_action(state: AgentState) -> AgentState:
    if state.get("approved"):
        state["final_output"] = f"Action executed: {state['pending_action']['type']}"
    else:
        state["final_output"] = "Action rejected by human reviewer"
    return state

def route_after_plan(state: AgentState) -> str:
    action = state.get("pending_action", {})
    if action.get("type") in ["send_email", "delete_record", "deploy"]:
        return "human_approval"
    return "execute_action"

workflow = StateGraph(AgentState)
workflow.add_node("plan_action", plan_action)
workflow.add_node("human_approval", human_approval)
workflow.add_node("execute_action", execute_action)
workflow.set_entry_point("plan_action")
workflow.add_conditional_edges("plan_action", route_after_plan)
workflow.add_edge("human_approval", "execute_action")
workflow.add_edge("execute_action", END)

graph = workflow.compile(checkpointer=MemorySaver())
result = graph.invoke({"messages": [], "pending_action": {}, "approved": False, "final_output": ""}, config={"configurable": {"thread_id": "1"}})
print("Result:", result["final_output"])
Trust calibration: Start with tight approval requirements and loosen them as you build confidence in the agent's behaviour. Track approval rates — if humans approve 99% of actions without changes, those actions can be auto-approved. If 30% get rejected or edited, tighten the prompt or add a validation step.

Define Risk Scoring

def score_action_risk(action_type: str, params: dict) -> int:
    """Return risk score 0-100. Higher = more dangerous."""
    base = {"search": 5, "read_file": 10, "write_file": 40, "send_email": 70, "delete": 90, "deploy": 95}
    score = base.get(action_type, 50)
    # Amplifiers
    if params.get("recipients_count", 1) > 10: score += 20
    if not params.get("reversible", True): score += 15
    if params.get("external", False): score += 10
    return min(score, 100)

test_actions = [
    ("search", {"query": "python docs"}, {}),
    ("write_file", {"path": "report.txt"}, {"reversible": True}),
    ("send_email", {"to": "boss@company.com"}, {"external": True, "reversible": False}),
    ("send_email", {"to": "all@company.com"}, {"recipients_count": 500, "external": False, "reversible": False}),
    ("delete", {"path": "/data/prod"}, {"reversible": False}),
]
for action, params, extra in test_actions:
    merged = {**params, **extra}
    risk = score_action_risk(action, merged)
    status = "🔴 BLOCK" if risk>80 else "🟡 REVIEW" if risk>40 else "🟢 AUTO"
    print(f"{status} ({risk:3d}) {action}: {params}")

Build the Email Drafting Agent

from langchain_openai import ChatOpenAI
from langchain.schema import HumanMessage, SystemMessage

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.3)

def draft_email(context: str) -> dict:
    """Generate an email draft from context."""
    response = llm([
        SystemMessage(content="You are an email writing assistant. Draft professional emails."),
        HumanMessage(content=f"Draft an email for this situation:\n{context}\nReturn JSON with keys: to, subject, body")
    ]).content
    import json, re
    match = re.search(r'\{.*\}', response, re.DOTALL)
    if match:
        return json.loads(match.group())
    return {"to": "recipient@example.com", "subject": "Follow-up", "body": response}

draft = draft_email("Inform the client that their project is delayed by 2 weeks due to technical issues.")
print("Draft email:")
print(f"  To: {draft.get('to')}")
print(f"  Subject: {draft.get('subject')}")
print(f"  Body: {draft.get('body', '')[:200]}")

Add the Approval Gate

def approval_gate(action_type: str, params: dict, auto_approve_threshold=40) -> bool:
    """Check risk and simulate human approval for high-risk actions."""
    risk = score_action_risk(action_type, params)
    if risk <= auto_approve_threshold:
        print(f"  ✅ Auto-approved (risk={risk})")
        return True
    print(f"\n  ⚠️  Human approval required (risk={risk})")
    print(f"  Action: {action_type}")
    print(f"  Params: {params}")
    # In production: send notification and wait. Here we simulate.
    user_input = input("  Approve? [y/n]: ").strip().lower()
    return user_input == "y"

def email_agent_with_gate(context: str) -> str:
    print(f"Step 1: Drafting email...")
    draft = draft_email(context)
    print(f"  Draft ready: '{draft.get('subject')}'")
    print(f"Step 2: Checking approval gate...")
    params = {"to": draft.get("to"), "external": True, "reversible": False}
    if approval_gate("send_email", params):
        return f"Email sent: {draft.get('subject')}"
    else:
        return "Email cancelled by human reviewer"

result = email_agent_with_gate("Thank the team for their work on Q3 results")
print(f"\nOutcome: {result}")

Audit Trail

import time

audit_log = []

def log_decision(action, risk, decision, reviewer="auto"):
    entry = {"timestamp": time.time(), "action": action, "risk": risk, "decision": decision, "reviewer": reviewer}
    audit_log.append(entry)
    print(f"  [AUDIT] {decision.upper()} — {action} (risk={risk}, reviewer={reviewer})")

# Simulate a session with multiple actions
actions_in_session = [
    ("search", {"query": "market trends"}),
    ("write_file", {"path": "draft.txt", "reversible": True}),
    ("send_email", {"to": "ceo@company.com", "external": True, "reversible": False}),
]
for action_type, params in actions_in_session:
    risk = score_action_risk(action_type, params)
    decision = "approved" if risk <= 40 else "pending_human"
    reviewer = "auto" if risk <= 40 else "human"
    log_decision(action_type, risk, decision, reviewer)

print(f"\nAudit log: {len(audit_log)} entries")
auto_rate = sum(1 for e in audit_log if e["reviewer"]=="auto") / len(audit_log) * 100
print(f"Auto-approval rate: {auto_rate:.0f}%")
